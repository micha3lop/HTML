if (isset($_COOKIE['visits'])) {
    $count = $_COOKIE['visits'] + 1;

    $count = 1;
}

?>

    <p>Welcome! You have visited this page <?php echo $count; ?> times.</p>
</body>
</html>

ðŸ§© 3. Remember Me Functionality with Cookies
<!DOCTYPE html>
<html>
<head><title>Remember Me</title></head>
<body>
<form method="post">
    Username: <input type="text" name="username" 
        value="<?php echo $_COOKIE['username'] ?? ''; ?>"><br><br>
    <input type="checkbox" name="remember"> Remember Me<br><br>
    <input type="submit" value="Login">
</form>

<?php
if (isset($_POST['remember'])) {
    setcookie('username', $_POST['username'], time() + 3600*24);
    echo "Welcome, " . $_POST['username'];
} else {
    echo "Welcome, " . ($_COOKIE['username'] ?? 'Guest');
}
?>
</body>
</html>

ðŸ§© 14. Simple Contact Form

<form method="post">

    $to = $_POST['to'];
    $sub = $_POST['sub'];
    $msg = $_POST['msg'];
    $headers = "From: user@domain.com";

    if (mail($to, $sub, $msg, $headers))
        echo "Mail Sent!";
    else
        echo "Mail Failed!";
}
?>
</body>
</html>
